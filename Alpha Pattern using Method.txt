// Method : 
//without parameter & without return 

using System;
class AlphaPattern{
  static void alpha()
    {
        char ctr='A';
        while(ctr<='E')
        {
            char ptr='A';
            while(ptr<=ctr)
            {
                Console.Write(ctr+" ");
                ptr++;
            }
            Console.WriteLine();
            ctr++;
        }
    }
    static void Main()
    {
        alpha();
    }
}
