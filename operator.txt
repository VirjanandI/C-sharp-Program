class PerformCalculation
{
    static int Calculation()
    {
      int result =0;
       System.Console.Write("Enter Number (Num1) : ");
        int num1 =int.Parse(System.Console.ReadLine());
       
        System.Console.Write("Enter Number (Num2) : ");
        int num2 = System.Convert.ToInt32(System.Console.ReadLine());
       
        System.Console.Write("1. Addition\n 2. Subtraction\n 3. Multiplication\n 4. Division "); 
        System.Console.Write("Enter Operator : ");
        char choice = System.Convert.ToChar(System.Console.ReadLine());

      switch (choice)
        {
            case '+':
                result = num1 + num2;
                break;

            case '-':
                result = num1 -num2;
                break;

            case '*':
                result = num1 * num2;
                break;

            case '/':
                result = num1 / num2;
                break;
        }
      return result;
    }
    static void Main()
    {
       
       System.Console.WriteLine(Calculation());
    }
}
