using System;
class StarPatternProgram
{
    static void StarPattern()
    {
        int ctr =1;
        while(ctr <= 5)
        {
            int ptr = 1;
            while(ptr <=5)
            {
                Console.Write(" * ");
                ptr++;
            }
            Console.WriteLine();
            ctr++;
        }
        
    }
    static void Main()
    {
        StarPattern();
    }
}
